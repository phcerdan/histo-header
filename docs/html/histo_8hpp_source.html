133 template <typename TData, typename Container>
134 TData variance_welford(const Container& xs)
135 {
136  unsigned long long N = 0;
137  TData M = 0, S = 0, Mprev = 0;
138  for(const auto & x : xs) {
139  ++N;
140  Mprev = M;
141  M += (x - Mprev) / N;
142  S += (x - Mprev) * (x - M);
143  }
144  return S / (N-1);
145 }

442  bool BalanceBreaksWithRange(std::vector<PRECI> &input_breaks, std::pair<PRECI,PRECI> input_range){
443  if (!CheckBreaksAreEquidistant(input_breaks)){
444  std::ostream_iterator<PRECI> out_it (std::cerr,", ");
445  std::copy ( input_breaks.begin(), input_breaks.end(), out_it );
446  throw histo_error("BalanceBreaksWithRange cannot be applied in NON Equidistant breaks");
447  }
448 
449  unsigned long int nbins = input_breaks.size() - 1;
450  PRECI width = input_breaks[1] - input_breaks[0];
451  // diff_low is > 0 when it does not reach range, and < 0 when it goes beyond.
452  PRECI diff_low = input_breaks[0] - input_range.first;
453  // diff_upper is < 0 when it does not reach range, and >0 when it goes beyond.
454  PRECI diff_upper = input_breaks[nbins] - input_range.second;
455  bool diff_low_isZero = isequalthan<PRECI>(diff_low, 0);
456  bool diff_upper_isZero = isequalthan<PRECI>(diff_upper, 0);
457 
458  if (diff_low_isZero && diff_upper_isZero) return false;
459 
460  // Put the first break in the range.first, and move all breaks accordingly.
461  if(!diff_low_isZero) ShiftBreaks(input_breaks, -diff_low);
462  CheckAndUpdateDiff(diff_upper, diff_upper_isZero, input_breaks[nbins], input_range.second);
463  if(diff_upper_isZero) return true;
464 
465  // Check if it is better to remove the last break than to add more.
466  // Biased to add more (1 is no bias).
467  double bias_to_add_bin = 0.8;
468  PRECI diff_upper_before = input_breaks[nbins - 1] - input_range.second;
469  if ((diff_upper_before < 0) && (diff_upper > 0 )
470  && ( fabs(diff_upper_before) < bias_to_add_bin * fabs(diff_upper) ))
471  {
472  nbins--;
473  input_breaks.pop_back();
474  PRECI width_to_expand = diff_upper_before/nbins;
475  ShrinkOrExpandBreaks(input_breaks, -width_to_expand);
476  }
477  CheckAndUpdateDiff(diff_upper, diff_upper_isZero, input_breaks[nbins], input_range.second);
478  if(diff_upper_isZero) return true;
479 
480  // If diff_upper is <0, add bins until reach the upper range.
481  while(diff_upper < 0){
482  nbins++;
483  input_breaks.push_back(input_range.first + nbins * width);
484  diff_upper = input_breaks[nbins] - input_range.second;
485  }
486  diff_upper_isZero = isequalthan<PRECI>(diff_upper, 0);
487  if (diff_upper_isZero) return true;
488 
489  // If diff_upper > 0.
490  // Shrink the width of the breaks, and calculate the new positions.
491  PRECI width_to_shrink = diff_upper / nbins;
492  ShrinkOrExpandBreaks(input_breaks, - width_to_shrink);
493  return true;
494  };

562 Histo<PRECI, PRECI> NormalizeByArea(const Histo<PRECI, PRECI_INTEGER> & input_histo)
563 {
564  // compute the area of each bin
565  double sum = 0.0;
566  for (size_t i = 0; i != input_histo.bins; ++i){
567  sum += input_histo.counts[i] * std::abs(input_histo.breaks[i+1] - input_histo.breaks[i]);
568  }

571  normalized.breaks = input_histo.breaks;

573  normalized.counts.resize(input_histo.bins);
574  std::transform(std::begin(input_histo.counts), std::end(input_histo.counts),
575  std::begin(normalized.counts), [&sum](const double &d){ return d / sum;} );
576  return normalized;
