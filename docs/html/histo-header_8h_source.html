00090 T variance_welford(const Container& xs)
00091 {
00092     unsigned long long N = 0;
00093     T M = 0, S = 0, Mprev = 0;
00094     for(auto x : xs) {
00095         ++N;
00096         Mprev = M;
00097         M += (x - Mprev) / N;
00098         S += (x - Mprev) * (x - M);
00099     }
00100     return S / (N-1);
